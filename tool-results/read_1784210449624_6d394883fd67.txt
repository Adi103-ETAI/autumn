   815→  arrangeNodes: () => {
   816→    // Tiered auto-layout:
   817→    //   Tier 0 — chat nodes with no incoming bus edges (roots)
   818→    //   Tier 1 — chat nodes reached from tier 0 via bus edges
   819→    //   Tier 2 — chat nodes reached from tier 1 (and so on)
   820→    //   Tier 3+ — non-chat nodes (terminal, screen, sticky, analytics, browser, remotion)
   821→    //             grouped by kind, in a grid below.
   822→    const nodes = get().nodes;
   823→    const edges = get().edges;
   824→    const busEdges = edges.filter((e) => e.kind === "bus");
   825→
   826→    const chatNodes = nodes.filter((n) => n.kind === "chat");
   827→    const otherNodes = nodes.filter((n) => n.kind !== "chat");
   828→
   829→    // Build adjacency for chat nodes via bus edges.
   830→    const incoming = new Map<string, string[]>();
   831→    const outgoing = new Map<string, string[]>();
   832→    for (const n of chatNodes) {
   833→      incoming.set(n.id, []);
   834→      outgoing.set(n.id, []);
   835→    }
   836→    for (const e of busEdges) {
   837→      if (incoming.has(e.target) && outgoing.has(e.source)) {
   838→        incoming.get(e.target)!.push(e.source);
   839→        outgoing.get(e.source)!.push(e.target);
   840→      }
   841→    }
   842→
   843→    // BFS tiers from roots (no incoming).
   844→    const tierOf = new Map<string, number>();
   845→    const roots = chatNodes.filter((n) => incoming.get(n.id)!.length === 0);
   846→    let queue: string[] = roots.map((n) => n.id);
   847→    roots.forEach((n) => tierOf.set(n.id, 0));
   848→    while (queue.length) {
   849→      const next: string[] = [];
   850→      for (const id of queue) {
   851→        const t = tierOf.get(id) ?? 0;
   852→        for (const child of outgoing.get(id) ?? []) {
   853→          if (!tierOf.has(child)) {
   854→            tierOf.set(child, t + 1);
   855→            next.push(child);
   856→          }
   857→        }
   858→      }
   859→      queue = next;
   860→    }
   861→    // Any disconnected chats go to tier 0.
   862→    for (const n of chatNodes) if (!tierOf.has(n.id)) tierOf.set(n.id, 0);
   863→
   864→    // Group chats by tier.
   865→    const tiers = new Map<number, string[]>();
   866→    for (const n of chatNodes) {
   867→      const t = tierOf.get(n.id) ?? 0;
   868→      if (!tiers.has(t)) tiers.set(t, []);
   869→      tiers.get(t)!.push(n.id);
   870→    }
   871→
   872→    const updated: AutumnNode[] = [];
   873→    const COL_W = 320;
   874→    const ROW_H = 220;
   875→    const PAD_X = 80;
   876→    const PAD_Y = 80;
   877→
   878→    // Place chats tier-by-tier (rows), spread horizontally within each tier.
   879→    const tierKeys = Array.from(tiers.keys()).sort((a, b) => a - b);
   880→    tierKeys.forEach((t) => {
   881→      const ids = tiers.get(t)!;
   882→      ids.forEach((id, i) => {
   883→        const n = nodes.find((x) => x.id === id)!;
   884→        updated.push({
   885→          ...n,
   886→          position: {
   887→            x: PAD_X + i * COL_W,
   888→            y: PAD_Y + t * ROW_H,
   889→          },
   890→        });
   891→      });
   892→    });
   893→
   894→    // Non-chat nodes: grouped by kind, in a grid below all chat tiers.
   895→    const NON_CHAT_ROW = PAD_Y + tierKeys.length * ROW_H + 40;
   896→    const byKind = new Map<string, typeof otherNodes>();
   897→    for (const n of otherNodes) {
   898→      if (!byKind.has(n.kind)) byKind.set(n.kind, []);
   899→      byKind.get(n.kind)!.push(n);
   900→    }
   901→    let col = 0;
   902→    for (const [, group] of byKind) {
   903→      group.forEach((n, i) => {
   904→        updated.push({
   905→          ...n,
   906→          position: {
   907→            x: PAD_X + col * (COL_W * 0.7),
   908→            y: NON_CHAT_ROW + i * 100,
   909→          },
   910→        });
   911→      });
   912→      col += 1;
   913→    }
   914→
   915→    set({ nodes: updated });
   916→    get().pushActivity({
   917→      kind: "commander_plan",
   918→      text: `Auto-arranged ${updated.length} node${updated.length === 1 ? "" : "s"} into a tiered layout.`,
   919→    });
   920→  },

  1476→// Execute a single Commander step. Returns the created node id if applicable.
  1477→function executeStep(
  1478→  get: () => AutumnStore,
  1479→  set: (partial: Partial<AutumnStore>) => void,
  1480→  step: { action: string; args: Record<string, unknown> },
  1481→  createdIds: string[],
  1482→  stepNum: number,
  1483→): string | null {
  1484→  const store = get();
  1485→  const { action, args } = step;
  1486→
  1487→  switch (action) {
  1488→    case "add_chat": {
  1489→      const personaArg = args.persona as string | undefined;
  1490→      const harnessArg = args.harness as string | undefined;
  1491→      const persona = personaArg
  1492→        ? PERSONA_BY_ID[resolvePersonaName(personaArg) ?? personaArg] ?? nextPersona()
  1493→        : nextPersona();
  1494→      const harness = harnessArg ?? persona.harness;
  1495→      const id = store.addNode({
  1496→        kind: "chat",
  1497→        name: args.name as string | undefined ?? persona.name,
  1498→        data: {
  1499→          harness,
  1500→          personaId: persona.id,
  1501→          model: defaultModelFor(harness),
  1502→          effort: "medium",
  1503→          permission: "ask",
  1504→          status: "idle",
  1505→          doing: "Standing by.",
  1506→          messages: [
  1507→            {
  1508→              id: `m-${nanoid(6)}`,
  1509→              role: "system",
  1510→              text: `${persona.name} online · ${harness} · ${persona.tagline}`,
  1511→              ts: Date.now(),
  1512→            },
  1513→          ],
  1514→        },
  1515→        position: randomPosition(),
  1516→      });
  1517→      return id;
  1518→    }
  1519→    case "add_agents": {
  1520→      const count = Math.max(1, Number(args.count ?? 1));
  1521→      const agent = args.agent as string | undefined;
  1522→      const surface = (args.surface as string | undefined) ?? "chat";
  1523→      let lastId: string | null = null;
  1524→      for (let i = 0; i < count; i++) {
  1525→        if (surface === "terminal") {
  1526→          lastId = store.addNode({ kind: "terminal" });
  1527→        } else {
  1528→          const persona = nextPersona();
  1529→          const harness = agent ?? persona.harness;
  1530→          lastId = store.addNode({
  1531→            kind: "chat",
  1532→            name: persona.name,
  1533→            data: {
  1534→              harness,
  1535→              personaId: persona.id,
  1536→              model: defaultModelFor(harness),
  1537→              effort: "medium",
  1538→              permission: "ask",
  1539→              status: "idle",
  1540→              doing: "Standing by.",
  1541→              messages: [
  1542→                {
  1543→                  id: `m-${nanoid(6)}`,
  1544→                  role: "system",
  1545→                  text: `${persona.name} online · ${harness} · ${persona.tagline}`,
  1546→                  ts: Date.now(),
  1547→                },
  1548→              ],
  1549→            },
  1550→            position: {
  1551→              x: 200 + i * 80 + Math.random() * 200,
  1552→              y: 120 + Math.random() * 280,
  1553→            },
  1554→          });
  1555→        }
  1556→      }
  1557→      return lastId;
  1558→    }
  1559→    case "add_terminal": {
  1560→      return store.addNode({ kind: "terminal", name: args.name as string | undefined });
  1561→    }
  1562→    case "add_screen": {
  1563→      const screenKind = (args.kind as "desktop" | "phone") ?? "desktop";
  1564→      return store.addNode({
  1565→        kind: "screen",
  1566→        name: args.name as string | undefined ?? `Preview · ${screenKind}`,
  1567→        data: {
  1568→          screenKind,
  1569→          name: args.name as string | undefined ?? "Preview",
  1570→          route: (args.route as string | undefined) ?? "/",
  1571→          url: "/preview",
  1572→        },
  1573→      });
  1574→    }
  1575→    case "add_note": {
  1576→      return store.addNode({
  1577→        kind: "sticky",
  1578→        name: "Note",
  1579→        data: {
  1580→          text: (args.text as string) ?? "New note",
  1581→          color: (args.color as "amber" | "rose" | "emerald" | "violet" | "cyan") ?? "amber",
  1582→        },
  1583→      });
  1584→    }
  1585→    case "add_browser": {
  1586→      const url = (args.url as string) ?? "https://example.com";
  1587→      return store.addNode({
  1588→        kind: "youtube",
  1589→        name: (args.name as string) ?? "Browser",
  1590→        data: { url, name: (args.name as string) ?? "Browser" },
  1591→      });
  1592→    }
  1593→    case "add_remotion": {
  1594→      return store.addNode({
  1595→        kind: "remotion",
  1596→        name: (args.name as string) ?? "Remotion",
  1597→        data: { name: "Remotion", crew: [], status: "idle" },
  1598→      });
  1599→    }
  1600→    case "connect_nodes": {
  1601→      const from = store.resolveNodeRef(args.from as string, createdIds);
  1602→      const to = store.resolveNodeRef(args.to as string, createdIds);
  1603→      if (from && to) store.connectNodes(from, to, "bus");
  1604→      return null;
  1605→    }
  1606→    case "disconnect_nodes": {
  1607→      const from = store.resolveNodeRef(args.from as string, createdIds);
  1608→      const to = store.resolveNodeRef(args.to as string, createdIds);
  1609→      if (from && to) store.disconnectNodes(from, to);
  1610→      return null;
  1611→    }
  1612→    case "rename_node": {
  1613→      const id = store.resolveNodeRef(args.id as string, createdIds);
  1614→      if (id) store.renameNode(id, args.name as string);
  1615→      return null;
  1616→    }
  1617→    case "move_node": {
  1618→      const id = store.resolveNodeRef(args.id as string, createdIds);
  1619→      if (id) store.moveNode(id, Number(args.x), Number(args.y));
  1620→      return null;
  1621→    }
  1622→    case "remove_node": {
  1623→      const id = store.resolveNodeRef(args.id as string, createdIds);
  1624→      if (id) store.removeNode(id);
  1625→      return null;
  1626→    }
  1627→    case "remove_nodes": {
  1628→      const ids = args.ids as string[] | undefined;
  1629→      const kind = args.kind as NodeKind | undefined;
  1630→      if (ids) ids.forEach((id) => store.removeNode(id));
  1631→      else if (kind) {
  1632→        get()
  1633→          .nodes.filter((n) => n.kind === kind)
  1634→          .forEach((n) => store.removeNode(n.id));

  1696→// ---- activity log persistence (debounced batch POST) ----
  1697→// Maintains a pending buffer of entries and flushes to /api/logs every ~1.5s
  1698→// to avoid one HTTP request per activity event.
  1699→let pendingLogs: ActivityEntry[] = [];
  1700→let logFlushTimer: ReturnType<typeof setTimeout> | null = null;
  1701→
  1702→function scheduleLogPersist(canvasId: string, entry: ActivityEntry) {
  1703→  pendingLogs.push(entry);
  1704→  if (logFlushTimer) clearTimeout(logFlushTimer);
  1705→  logFlushTimer = setTimeout(async () => {
  1706→    const batch = pendingLogs.slice();
  1707→    pendingLogs = [];
  1708→    logFlushTimer = null;
  1709→    if (batch.length === 0) return;
  1710→    try {
  1711→      await fetch("/api/logs", {
  1712→        method: "POST",
  1713→        headers: { "Content-Type": "application/json" },
  1714→        body: JSON.stringify({
  1715→          canvasId,
  1716→          entries: batch.map((e) => ({
  1717→            kind: e.kind,
  1718→            text: e.text,
  1719→            nodeId: e.nodeId ?? "commander",
  1720→            meta: e.meta,
  1721→            ts: e.ts,
  1722→          })),
  1723→        }),
  1724→      });
  1725→    } catch {
  1726→      /* ignore — best-effort persistence */
  1727→    }
  1728→  }, 1500);
  1729→}
