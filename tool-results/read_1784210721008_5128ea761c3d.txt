  1162→        incoming.get(e.target)!.push(e.source);
  1163→        outgoing.get(e.source)!.push(e.target);
  1164→      }

  1169→    const roots = chatNodes.filter((n) => incoming.get(n.id)!.length === 0);
  1170→    let queue: string[] = roots.map((n) => n.id);
  1171→    roots.forEach((n) => tierOf.set(n.id, 0));
  1172→    while (queue.length) {

  1176→        for (const child of outgoing.get(id) ?? []) {
  1177→          if (!tierOf.has(child)) {
  1178→            tierOf.set(child, t + 1);
  1179→            next.push(child);
  1180→          }
  1181→        }
  1182→      }
  1183→      queue = next;
  1184→    }

  1193→      tiers.get(t)!.push(n.id);
  1194→    }

  1202→    // Place chats tier-by-tier (rows), spread horizontally within each tier.
  1203→    const tierKeys = Array.from(tiers.keys()).sort((a, b) => a - b);
  1204→    tierKeys.forEach((t) => {
  1205→      const ids = tiers.get(t)!;
  1206→      ids.forEach((id, i) => {

  1208→        updated.push({
  1209→          ...n,
  1210→          position: {
  1211→            x: PAD_X + i * COL_W,

  1213→          },
  1214→        });
  1215→      });

  1218→    // Non-chat nodes: grouped by kind, in a grid below all chat tiers.
